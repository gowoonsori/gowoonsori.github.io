template<typename T>

bool BST<T>::search_loop(T item) {
    Node<T>* s = this->root;
    while (s != NULL) {
        if (s->key == item) return true;   //keyÍįÄ ž°īžě¨ŪēúŽč§Ž©ī true
        else if (s->key > item) s = s->LNode; //keyÍįÄ itemŽ≥īŽč§ ŪĀ¨Žč§Ž©ī žôľž™ĹžúľŽ°ú žĚīŽŹô
        else s = s->RNode;                //keyÍįÄ itemŽ≥īŽč§ žěĎŽč§Ž©ī žė§Ž•łž™ĹžúľŽ°ú žĚīŽŹô
    }
    return false;

}

template<typename T>
void BST<T>::add(Node<T>* node) {
    T item = node->key;
    /*ŪäłŽ¶¨ÍįÄ ŽĻĄžĖīžěąŽč§Ž©ī*/
    if (this->root == NULL) {
        this->root = node;
        return;
    }
    /*ÍįížĚī ž°īžě¨ŪēėŽ©ī*/
    if (search_loop(item)) {
        std::cout << "ÍįížĚī žĚīŽĮł ž°īžě¨Ūē©ŽčąŽč§" << std::endl;
        return;
    }
    /*item žāĹžěÖ*/
    Node<T>* s = this->root;

    while (1) {
        /*itemžĚī keyÍįíŽ≥īŽč§ ŪĀ¨Žč§Ž©ī*/
        if (s->key < item) {
            if (s->RNode == NULL) {
                s->RNode = node;
                return;
            }
            s = s->RNode;
        }
        else {
            if (s->LNode == NULL) {
                s->LNode = node;
                return;
            }
            s = s->LNode;
        }
    }
}

template<typename T>
void BST<T>::Delete(T item) {
    Node<T>* t = this->root;
    Node<T>* parent = NULL;
    Node<T>* child, * succ, * succ_p;

    /*keyÍįížĚĄ žįĺÍĪįŽāė žóÜŽč§Ž©ī break*/
    while (t != NULL && t->key != item) {
        parent = t;
        t = (item < parent->key) ? parent->LNode : parent->RNode;
    }

    /*-----ŪÉźžÉČ žĘÖŽ£Ć------*/
    /*ÍįížĚī žóÜŽč§Ž©ī*/
    if (t == NULL) {
        std::cout << "ŪēīŽčĻ ÍįížĚī ž°īžě¨ ŪēėžßÄ žēäžäĶŽčąŽč§." << std::endl;
        return;
    }

    /*žěźžčĚŽÖłŽďúÍįÄ žóÜŽč§Ž©ī*/
    if (t->RNode == NULL && t->LNode == NULL) {
        if (parent != NULL) {
            /*Ž∂ÄŽ™®ŽÖłŽďúžĚė žôľž™Ĺžóź žěąŽč§Ž©ī*/
            if (parent->LNode == t) parent->LNode = NULL;
            else parent->RNode = NULL;
            delete(t);
        }
        else {
            delete(t);
            this->root = NULL;
        }
    }

    /*1ÍįúžĚė žěźžčĚžĚī žěąŽč§Ž©ī */
    else if (t->RNode == NULL || t->LNode == NULL) {
        child = (t->LNode == NULL) ? t->RNode : t->LNode;
        if (parent != NULL) {
            if (parent->LNode == t) parent->LNode = child;
            else parent->RNode = child;
            delete(t);
        }
        /*žā≠ž†úŪē† ŽÖłŽďúÍįÄ rootŽĚľŽ©ī*/
        else {
            delete(t);
            this->root = child;
        }
    }

    /*2ÍįúžĚė žěźžčĚžĚī žěąŽč§Ž©ī žė§Ž•łž™Ĺ childžĚė ÍįÄžě• žěĎžĚÄ ÍįížĚĄ attach*/
    else {
        succ_p = t;
        succ = t->RNode;

        /*žė§Ž•łž™Ĺ childžĚė ÍįÄžě• žěĎžĚÄ Íįí žįĺÍłį*/
        /*succ_p == ÍįÄžě• žěĎžĚÄ ÍįížĚė Ž∂ÄŽ™® ŽÖłŽďú*/
        /*succ ŽäĒ ÍįÄžě• žěĎžĚÄ ÍįížĚė ŽÖłŽďú*/
        while (succ->LNode != NULL) {
            succ_p = succ;
            succ = succ->LNode;
        }

        /*žā≠ž†úŪēėŽ†§ŽäĒ ŽÖłŽďúžĚė žė§Ž•łž™Ĺ childžĚė leftnodeÍįÄ žěąŽč§Ž©ī
        succ_p leftnodežĚė rightnodeŽ•ľ
        succ_pžĚė žôľž™Ĺžóź žóįÍ≤į*/
        if (succ_p->LNode == succ) {
            succ_p->LNode = succ->RNode;
        }
        /*žā≠ž†úŪēėŽ†§ŽäĒ ŽÖłŽďúžĚė žė§Ž•łž™ĹchildžĚė left nodeÍįÄ žóÜŽč§Ž©ī
        žė§Ž•łž™Ĺ childžĚė right nodeŽ•ľ žā≠ž†úŪē† ŽÖłŽďúžĚė žė§Ž•łž™Ĺžóź žóįÍ≤į*/
        else succ_p->RNode = succ->RNode;

        /*žā≠ž†úŪē† ŽÖłŽďúžôÄ ÍĶźŪôė ŪõĄ žā≠ž†ú*/
        t->key = succ->key;
        t = succ;
        delete(t);
    }
    return;


}
