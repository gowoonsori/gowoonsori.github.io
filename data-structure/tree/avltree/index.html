    BST (이진 탐색 트리)를 기반으로 둔 Tree.

    어떤 노드를 기준으로 하더라도 왼쪽자식의 깊이와 오른쪽 자식의 깊이 차이가 1을 넘지 않는 트리



    📌 용어 개념 정리

    • 균형치 (Balance factor) : 자식노드의 깊이 차이 ( 왼쪽 서브트리의 높이 – 오른쪽 서브트리의 높이 )
      BF는 -1, 0, 1이 기준이며, 이 범위를 벗어난다면, 그 트리의 균형은 깨진것다.



    📌 특징

    • BST의 모든 특징을 갖는다.

    • 모든 노드를 기준으로 왼쪽자식의 깊이와 오른쪽 자식의 깊이 차이가 1을 넘지 않는다.

    • AVL Tree가 Red Black Tree보다 더 엄격하기 때문에 더 빠른 조회가 가능하나 삽입 / 삭제가 RB Tree에 비해 느리다.

    • RB Tree는 Red/Black을 표현할 때 1 or 0 1비트만 있어도 되나 AVL Tree는 깊이를 저장할 int타입의 저장이 필요하다.

    • 삽입, 삭제, 탐색의 시간복잡도는 O(log n)



    📌 사용 예

    • 더 빠른 검색이 필요한 DB에 사용

    Search ( 탐색 )

    1AVL Tree도 BST의 일종이기 때문에 탐색의 밥법은 일반적인 Bianry Tree의 탐색 방법과 다르지 않다.
2
3찾는 값이 해당 노드보다 작다면 왼쪽으로 크다면 오른쪽으로 내려가며 값을 탐색

  6*

  8*       단순 연결 리스트로 구현

 10*       class AVLTree
 11*
 12*       변수 :   root => root node

 14*       생성자 : RBTREE =>  root 를 null로 초기화
 15*
 16*       함수 :   IsKey => key값이 있는지 검사하는 함수

 19*               Delete => 재귀를 이용한 삭제 함수 (최종적을 root를 return)
 20*               Balancing => 삽입 / 삭제후 BF 검사하여 규칙깨졌을시 재조정 함수
 21*               Transplant => 삭제 시 이용하며, 삭제할 노드의 자식 노드를 부모노드에 연결해주는 함수
 22*
 23*               getHeight(x) => x의 높이 getter
 24*               getBalanceBacotr(x) => x의 BF 계산하여 return
 25*               RotateRight(x) => x기준 오른쪽으로 회전
 26*               RotateLeft(x) => x기준 왼쪽으로 회전
 27*
 28*               Inorder,Preorder,Postorder => 순회 함수
 29*               tree_minimum(x), tree_maximum(x) => 노드 x 기준으로 가장 왼쪽, 오른쪽 return 함수
 30*
 31*               DisplayMenu, SelectMenu => 초기 Menu판 print 함수
 32*               Insert_helper,Delete_helper,order_helper,print_helper => 각각 이벤트 수행시 입력받고 조건 에러 처리 위한 함수 와 tree print 해주는 함수
 33*
 34*        Balancing에서 각 case에 대한 설명은 github에 적어 놓았다.
 35*
 36* 작성자 : gowoonsori

 40#include <algorithm>  // max() 함수 이용
 41#include <iostream>
 42
 43struct node {
 44    int key;
 45    node *left = nullptr;
 46    node *right = nullptr;
 47    int height = 1;
 48};
 49typedef node *NodePtr;
 50
 51class AVLTREE {
 52   private:
 53    NodePtr root;  //루트 노드
 54
 55    //key값이 있는지 없는지 검사 있으면 pointer 값, 없으면 nullptr
 56    NodePtr IsKey(int item) {
 57        NodePtr t = root;
 58
 59        /*key값을 찾거나 없다면 break*/
 60        while (t != nullptr && t->key != item) {
 61            t = (item < t->key) ? t->left : t->right;
 62        }
 63        return t;
 64    }
 65
 66    /*새로운 key 삽입함수로 root노드 반환*/
 67    NodePtr Insert(NodePtr r, int item) {
 68        /*새로운 노드 삽입*/
 69        if (r == nullptr) {
 70            NodePtr z = new node;
 71            z->key = item;
 72            r = z;
 73            return r;
 74        } else if (r->key < item) {  //item이 key값보다 크다면 오른쪽으로 이동
 75            r->right = Insert(r->right, item);
 76        } else {  //item이 key값보다 작다면 왼쪽으로 이동
 77            r->left = Insert(r->left, item);
 78        }
 79        r->height = std::max(getHeight(r->left), getHeight(r->right)) + 1;
 80        Balancing(r, item);  //새로운 노드가 추가되었으므로 재귀적으로 부모노드들 높이 1증가 시켜주고
 81                             //Balace Factor 측정하여 2이상이라면 재조정함수
 82        return r;
 83    }
 84
 85    /*key 삭제 함수*/
 86    NodePtr Delete(NodePtr r, int item) {
 87        if (r->key > item && r->left != nullptr) {
 88            r->left = Delete(r->left, item);
 89        } else if (r->key < item && r->right != nullptr) {
 90            r->right = Delete(r->right, item);
 91        } else if (r->key == item) {
 92            Transplant(r);
 93        }
 94        /*root를 지운게 아니라면*/
 95        if (r != nullptr) {
 96            r->height = std::max(getHeight(r->left), getHeight(r->right)) + 1;
 97            Balancing(r, item);
 98        }
 99
100        return r;
101    }
102
103    /* balance Factor 측정후 재조정*/
104    void Balancing(NodePtr &r, int item) {
105        int bF = getBalanceFactor(r);
106
107        //case 1 (left left)
108        if (bF > 1 && item < r->left->key) {
109            r = RotateRight(r);
110        }
111        //case 2 (left right)
112        else if (bF > 1 && item > r->left->key) {
113            r->left = RotateLeft(r->left);
114            r = RotateRight(r);
115        }
116        //case 3 (right right)
117        else if (bF < -1 && item > r->right->key) {
118            r = RotateLeft(r);
119        }
120        //case 4 ( right left)
121        else if (bF < -1 && item < r->right->key) {
122            r->right = RotateRight(r->right);
123            r = RotateLeft(r);
124        }
125    }
126
127    /* 삭제시 자식 이식 함수*/
128    void Transplant(NodePtr &x) {
129        NodePtr y = x;  //y를 통해 노드 삭제
130
131        if (x->left == nullptr) {
132            x = x->right;
133        } else if (x->right == nullptr) {
134            x = x->left;
135        } else {
136            NodePtr z = x->right;  //z : 삭제할 x의 다음으로 가장 작은 수
137            NodePtr pZ = x;        //p[z] : z의 부모 노드
138
139            /* 오른쪽 자식중 가장 작은 값*/
140            while (z->left != nullptr) {
141                pZ = z;
142                z = z->left;
143            }
144
145            x->key = z->key;  //successor과 key값 교환
146
147            /*오른쪽 자식이 가장 작다면*/
148            if (pZ == x) {
149                x->right = z->right;  // z의 오른쪽 자식 붙여주기
150            } else {
151                pZ->left = z->right;  // 오른쪽 자식의 왼쪽 자식이 있다면
152            }                         // 그 z(successor)의 오른쪽 자식 p[z]의 왼쪽에 붙여주기
153            y = z;                    //값 삭제를 위해 y <- z;
154        }
155        delete y;
156    }
157    /*높이 getter */
158    int getHeight(NodePtr r) {
159        if (r == nullptr)
160            return 0;
161        else
162            return r->height;
163    }
164    /*좌우 자식 깊이 비교하여 Balnace Factor get*/
165    int getBalanceFactor(NodePtr r) {
166        return getHeight(r->left) - getHeight(r->right);
167    }
168
169    /*x를 중심으로 왼쪽으로 회전*/
170    NodePtr RotateLeft(NodePtr x) {
171        NodePtr y = x->right;
172        x->right = y->left;
173        y->left = x;
174
175        //위치가 바뀌었으므로 높이 재조정
176        x->height = std::max(getHeight(x->left), getHeight(x->right)) + 1;
177        y->height = std::max(getHeight(y->left), getHeight(y->right)) + 1;
178
179        return y;
180    }
181    /*y를 중심으로 오른쪽으로 회전*/
182    NodePtr RotateRight(NodePtr y) {
183        NodePtr x = y->left;
184        y->left = x->right;
185        x->right = y;
186
187        //위치가 바뀌었으므로 높이 재조정
188        y->height = std::max(getHeight(y->left), getHeight(y->right)) + 1;
189        x->height = std::max(getHeight(x->left), getHeight(x->right)) + 1;
190
191        return x;
192    }
193
194    /*show tree*/
195    void print_helper(NodePtr root, std::string indent, bool last) {
196        // print the tree structure on the screen
197        if (root != nullptr) {
198            std::cout << indent;
199            if (last) {
200                std::cout << "R----";
201                indent += "     ";
202            } else {
203                std::cout << "L----";
204                indent += "|    ";
205            }
206
207            int height = std::max(getHeight(root->left), getHeight(root->right)) + 1;
208            std::cout << root->key << " (" << height << ")" << std::endl;
209            print_helper(root->left, indent, false);
210            print_helper(root->right, indent, true);
211        }
212    }
213
214    /*중위순회*/
215    void Inorder(NodePtr target) {
216        if (target == nullptr)
217            return;
218        Inorder(target->left);
219        std::cout << target->key << " ";
220        Inorder(target->right);
221    }
222    /*후위순회*/
223    void Postorder(NodePtr target) {
224        if (target == nullptr)
225            return;
226        Postorder(target->left);
227        Postorder(target->right);
228        std::cout << target->key << " ";
229    }
230    /*전위순회*/
231    void Preorder(NodePtr target) {
232        if (target == nullptr)
233            return;
234        std::cout << target->key << " ";
235        Preorder(target->left);
236        Preorder(target->right);
237    }
238
239   public:
240    AVLTREE() {
241        this->root = nullptr;
242    }
243    //최솟값 찾기
244    NodePtr tree_minimum(NodePtr x) {
245        while (x->left != nullptr) {
246            x = x->left;
247        }
248        return x;
249    }
250    //최댓값 찾기
251    NodePtr tree_maximum(NodePtr x) {
252        while (x->right != nullptr) {
253            x = x->right;
254        }
255        return x;
256    }
