    ⚡ 문제 링크 : 프로그래머스



    ✔ 문제

    출발지점부터 distance만큼 떨어진 곳에 도착지점이 있습니다. 그리고 그사이에는 바위들이 놓여있습니다.
바위 중 몇 개를 제거하려고 합니다.

예를 들어, 도착지점이 25만큼 떨어져 있고, 바위가 [2, 14, 11, 21, 17] 지점에 놓여있을 때 바위 2개를 제거하면 출발지점, 도착지점, 바위 간의 거리가 아래와 같습니다.

    제거한 바위의 위치 각 바위 사이의 거리 거리의 최솟값
    [21, 17] [2, 9, 3, 11] 2
    [2, 21] [11, 3, 3, 8] 3
    [2, 11] [14, 3, 4, 4] 3
    [11, 21] [2, 12, 3, 8] 2
    [2, 14] [11, 6, 4, 4] 4 

    위에서 구한 거리의 최솟값 중에 가장 큰 값은 4입니다.

출발지점부터 도착지점까지의 거리 distance, 바위들이 있는 위치를 담은 배열 rocks, 제거할 바위의 수 n이 매개변수로 주어질 때,

바위를 n개 제거한 뒤 각 지점 사이의 거리의 최솟값 중에 가장 큰 값을 return 하도록 solution 함수를 작성해주세요.

    ❗ 제한사항

    1. 도착지점까지의 거리 distance는 1 이상 1,000,000,000 이하입니다.
2. 바위는 1개 이상 50,000개 이하가 있습니다.
3. n 은 1 이상 바위의 개수 이하입니다.

    ❗ 입출력 예

    distance rocks n return
    25 [2, 14, 11, 21, 17] 2 4



    ✔ 문제 풀이


    📚 알고리즘

    문제 유형이 이분 탐색이라는 것을 알고 풀이를 진행 해서 그나마 다행이었지 아무것도 모르고 풀었다면 이분탐색으로 접근을 못했을 것 같은 문제였다.

    우선 문제 접근은 바위 사이의 거리 x를 정해 놓고 x보다 작은 거리를 갖는 바위는 제거를 다 하고, 이 제거한 바위의 개수가 문제의 n보다 크다면 x를 줄이고, n보다 작다면 x를 늘려가면서 x값을 찾는 방법으로 접근했습니다.

    이 문제를 풀면서 이분탐색 쪽이 많이 약하구나를 느꼈고 이분탐색의 유형을 많이 풀어봐야할 것 같다.


    📃 코드 ( 주석 없는 코드 )

    #include <bits/stdc++.h>

using namespace std;

int solution(int distance, vector<int> rocks, int n)
{
    int answer = 1;
    rocks.push_back(0);
    rocks.push_back(distance);
    sort(rocks.begin(), rocks.end());
    int l = 1, r = distance;

    while (l <= r)
    {
        int mid = (l + r) / 2;
        int prev_rock = 0;
        int remove = 0;

        for (int i = 1; i < rocks.size(); i++)
        {
            int gap = rocks[i] - prev_rock;

            if (gap < mid)
            {
                remove++;
            }
            else
            {
                prev_rock = rocks[i];
            }
        }

        if (remove > n)
        {
            r = mid - 1;
        }
        else
        {
            l = mid + 1;
            answer = mid;
        }
    }

    return answer;
}



    📃 코드 ( 주석 있는 코드 )

    ( 접기 / 펼치기 ) 
    #include <bits/stdc++.h>

using namespace std;

int solution(int distance, vector<int> rocks, int n)
{
    int answer = 1;
    rocks.push_back(0);         //시작 지점부터 첫번째 바위까지의 거리와
    rocks.push_back(distance);  //마지막 바위부터 distance까지 거리를 구하기 위해
    sort(rocks.begin(), rocks.end());  //입력이 정렬되어있지 않기 때문에 sort

    int l = 1, r = distance;
    while (l <= r)
    {
        int x = (l + r) / 2;      //바위사이의 거리 x
        int prev_rock = 0;        //앞에 위치한 바위의 위치를 저장할 변수
        int remove = 0;           // 제거한 바위의 개수 count할 변수

        for (int i = 1; i < rocks.size(); i++)
        {
            int gap = rocks[i] - prev_rock;     //바위 사이의 거리 측정

            /* x보다 작다면 바위를 제거한 것으로 간주*/
            if (gap < x)
            {
                remove++;
            }

            /*x보다 크다면  앞에 위치한 바위 위치 저장

              x보다 작을때는 prev_rock에 담지 않는 이유는 해당 바위는 제거한 것으로
              간주했기 때문이다.

              예를 들어 0 2 11 15 의 case에서 x가 현재 8인 상태에서 반복문을 돌고 있다고 한다면
              첫번째 바위의 거리는 2가 되므로 첫번째 바위를 제거하고 두번째 바위의 거리를 측정하게 되는 데  이때 첫번째 바위는 제거한 상태이므로 11 - 2 = 9가 아닌 두번째 바위까지의 거리는 그 전의 바위 정보인 11 - 0 = 11이 되어야 하기 때문.  */
            else
            {
                prev_rock = rocks[i];
            }
        }

        /* 제거한 바위의 개수가 n보다 많다면 x값 줄이기*/
        if (remove > n)
        {
            r = x - 1;
        }
        /* 제거한 바위의 개수가 n보다 적다면 x값 늘리기*/
        else
        {
            l = x + 1;
            answer = x;
        }
    }

    return answer;
}
