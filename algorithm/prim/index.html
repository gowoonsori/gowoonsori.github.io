    Pesudo Code

    MST-Prim(G, w, r)
    Q = V[G];
    for each u вҲҲ Q
        key[u] = INFINITY;
    key[r] = 0;
    p[r] = NULL;
    while (Q not empty)
        u = ExtractMin(Q);
        for each v вҲҲ Adj[u]
            if (v вҲҲ Q and w(u,v) < key[v])  //Decrease-key
                p[v] = u;
                key[v] = w(u,v);



    кө¬нҳ„ л°©лІ•

    1. vertexл“Өмқҳ keyк°’мқ„ InfinityлЎң мҙҲкё°нҷ”н•ңлӢӨ.

    2. start vertexмқҳ keyк°’мқ„ 0мңјлЎң мҙҲкё°нҷ”н•ңлӢӨ. (м–ҙл–Ө vertexлҘј м„ нғқн•ҳлҚ”лқјкі  MSTк°Җ лӮҳмҳЁлӢӨ.)

    3. нҳ„мһ¬ vertexм—җ мқём ‘н•ң vertexл“Ө мӨ‘ м„ нғқн•ҳм§Җ м•Ҡм•ҳкі , к°ҖмһҘ vertexмқҳ keyк°’мқҙ мһ‘мқҖ vertexмқ„ м°ҫлҠ”лӢӨ. (exract-min = мөңмҶҢк°’ м¶”м¶ң)

    4. нҳ„мһ¬ vertexлҘј м„ нғқн•ңлӢӨ.

    5. мқём ‘н•ң vertexмӨ‘ vertexмқҳ keyк°’ліҙлӢӨ к°„м„ мқҳ к°ҖмӨ‘м№ҳк°Җ лҚ” мһ‘лӢӨл©ҙ keyк°’мқ„ к°ҖмӨ‘м№ҳлЎң к°ұмӢ  н•ңлӢӨ. (decrease -key)

    6. мқём ‘н•ң vertexмӨ‘ м„ нғқн•ҳм§Җ м•Ҡм•ҳкі , к°ҖмһҘ vertexмқҳ keyк°’мқҙ мһ‘мқҖ vertexлҘј кё°мӨҖмңјлЎң 3лІҲл¶Җн„° лӢӨмӢң л°ҳліөн•ңлӢӨ.

    7. лӘЁл“  vertexк°Җ м„ нғқлҗҳм—ҲлӢӨл©ҙ лҒқлӮңлӢӨ.


    мқём ‘ н–үл ¬

    int v = -1;              //мқём ‘ vertexмӨ‘ к°ҖмһҘ мһ‘мқҖ к°ҖмӨ‘м№ҳлҘј к°–лҠ” vertex
int min_key = INFINITY;  //мқём ‘ vertexмӨ‘ к°ҖмһҘ мһ‘мқҖ к°ҖмӨ‘м№ҳ

for (int j = 0; j < V; j++) {
  if (!selected[j] && (min_key > vertex_key[j])) {
    v = j;

  }

    for (int j = 0; j < V; j++) {

    5лІҲ л°©лІ•мңјлЎң мқём ‘н•ң vertexмӨ‘ vertexмқҳ keyк°’ліҙлӢӨ к°„м„ мқҳ к°ҖмӨ‘м№ҳк°Җ лҚ” мһ‘лӢӨл©ҙ keyк°’мқ„ к°ҖмӨ‘м№ҳлЎң к°ұмӢ  н•©лӢҲлӢӨ.


    мқём ‘ лҰ¬мҠӨнҠё

    std::set<II> q;  //мқҙм§„нһҷмңјлЎң queue л§Ңл“Өкё° ( setмқҖ red-black treeлЎң л§Ңл“Өм–ҙм§җ )
auto u = q.begin();  // extract-min

    мқём ‘ лҰ¬мҠӨнҠёлҠ” мқём ‘н•ң vertexмқҳ к°ҖмӨ‘м№ҳлҘј priority queueлҘј нҶөн•ҙ м ҖмһҘн•ҳкё°л•Ңл¬ём—җ 3лІҲ л°©лІ•мқҳ exract-minмқҖ л§Ём•һм—җм„ң popмқ„ мӢңмјң м°ҫм•„мЈјл©ҙ лҗ©лӢҲлӢӨ.

    /*selectн•ң vertexмҷҖ мқём ‘н•ң к°„м„ мқё e*/
for (auto e : adjList[нҳ„мһ¬ vertex]) {
/* м„ нғқлҗҳм§Җ м•ҠмқҖ vertexмқҙкі  н•ҙлӢ№ vertexмқҳ keyк°’кіј edgeмқҳ costлҘј л№„көҗн•ҙ costк°Җ лҚ” мһ‘лӢӨл©ҙ*/
  if (!selected[мқём ‘н•ң vertex] && vertex_key[мқём ‘н•ң vertex] > к°ҖмӨ‘м№ҳ) {
    q.erase({vertex_key[мқём ‘н•ң vertex], e.second});  //к°ҷмқҖ vertexлЎң н–Ҙн•ҳлҠ” к°„м„ мӨ‘ weightк°Җ лҚ” мһ‘мқҖ к°„м„ мқҙ мһҲлӢӨл©ҙ к·ё м „ к°„м„ мқҖ мӮӯм ң
    vertex_key[мқём ‘н•ң vertex] = к°ҖмӨ‘м№ҳ;  // vertex keyк°’ к°ұмӢ 
    q.insert({к°ҖмӨ‘м№ҳ, мқём ‘н•ң vertex});   //нҒҗм—җ мӮҪмһ…
  }
}

    5лІҲ л°©лІ•мңјлЎң мқём ‘н•ң vertexмӨ‘ vertexмқҳ keyк°’ліҙлӢӨ к°„м„ мқҳ к°ҖмӨ‘м№ҳк°Җ лҚ” мһ‘лӢӨл©ҙ keyк°’мқ„ к°ҖмӨ‘м№ҳлЎң к°ұмӢ  н•ҳлҠ” л°©лІ•мқҖ м•„лһҳмҷҖ к°ҷмқҙ мқём ‘н•ң к°„м„ мқҳ к°ңмҲҳл§ҢнҒј мҲҳн–үн•©лӢҲлӢӨ.



    мӢңк°„ ліөмһЎлҸ„

    мӢңк°„ліөмһЎлҸ„лҠ” мҙҲкё°нҷ”н•ҳлҠ”лҚ° O(|V|), MSTкі„мӮ°н•ҳлҠ”лҚ° O(|V|) * T(extract-min) (к°ҖмһҘ м ҒмқҖ к°’ м¶”м¶ңн•ҳлҠ”лҚ° кұёлҰ°мӢңк°„) + O(|E|) * T(decrease-key) ( keyк°’ ліҖкІҪн•ҳлҠ”лҚ° кұёлҰ¬лҠ” мӢңк°„ )мқҙкё° л•Ңл¬ём—җ priority-queueлҘј м–ҙл–»кІҢ кө¬нҳ„н–ҲлҠ”м§Җм—җ л”°лқј мӢңк°„ліөмһЎлҸ„к°Җ лӢ¬лқјм§‘лӢҲлӢӨ.

    мқјл°ҳ л°°м—ҙлЎң кө¬нҳ„н–Ҳмқ„ кІҪмҡ° T(extract)к°Җ O(|V|), T(decrease)лҠ” O(1)л§ҢнҒј кұёл Ө мҙқ O(|V^2|)мқҙ кұёлҰҪлӢҲлӢӨ.

    binary heap(мқҙм§„ нһҷ)мңјлЎң кө¬нҳ„н•ҳл©ҙ T(extract)к°Җ O(lgV), T(decrease)лҠ” O(lgV)л§ҢнҒј кұёл Ө мҙқ O(VlgV) + O(ElgV) мқҙкё° л•Ңл¬ём—җ O((E+V)lgV)л§ҢнҒј кұёлҰ°кі  л¬ҙл°©н–Ҙ к·ёлһҳн”„мқјл•Ң Eмқҳ мөңмҶҢк°’мқҖ V-1лЎң кұ°мқҳ лҢҖл¶Җл¶„мқҙ |E| > |V| мқҙлҜҖлЎң O(|E|lg|V|)лқјкі  н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

    fibonacci heapмңјлЎң кө¬нҳ„н•ҳл©ҙ decrease-keyмқҳ мӢңк°„мқ„ мўҖлҚ” мӨ„мқј мҲҳ мһҲлҠ”лҚ° decrease-keyмӢңк°„мқҙ O(1)л§ҢнҒј кұёлҰ¬кё° л•Ңл¬ём—җ O(E+VlgV)лқјкі  н•  мҲҳ мһҲмҠөлӢҲлӢӨ.
    O(E) or O(VlgV) мқё мқҙмң лҠ” мөңм•…мқҳ кІҪмҡ°м—җ EлҠ” O(V^2)мқҙкё° л•Ңл¬ёмһ…лӢҲлӢӨ.



    кө¬нҳ„ мҪ”л“ң

typedef struct edge {
    int src;     //м¶ңл°ң vertex
    int dest;    //лҸ„м°© vertex
    int weight;  //к°ҖмӨ‘м№ҳ(л№„мҡ©)
} edge;

class Graph {
   private:
    edge e;

   public:
    Graph(int src = 0, int dest = 0, int weight = 0) {
        this->e.src = src;
        this->e.dest = dest;
        this->e.weight = weight;
    }
    int getSrc() { return this->e.src; }
    int getDest() { return this->e.dest; }
    int getWeight() { return this->e.weight; }
};

int prim_adjList_heap(std::vector<Graph> &g, std::vector<std::vector<II>> adjList, int start) {
    int sum = 0;
    std::set<II> q;                               //мқҙм§„нһҷмңјлЎң queue л§Ңл“Өкё° ( setмқҖ red-black treeлЎң л§Ңл“Өм–ҙм§җ )
    std::vector<int> vertex_key(V, INFINITY);     // vertexмқҳ мөңмҶҢ weightк°’ кі„мӮ°
    std::vector<bool> selected(g.size(), false);  //м„ нғқлҗң vertexмқёк°Җ

    vertex_key[start] = 0;
    q.insert(II(0, start));  //мӢңмһ‘ л…ёл“ң к°ҖмӨ‘м№ҳ 0мңјлЎң мӢңмһ‘
    std::cout << "\nroute";

    /*vertex мҲҳл§ҢнҒј л°ҳліөн•ңлӢӨ
     whileлҢҖмӢ  for(int i=0; i < V ; i++)лЎң н•ҙлҸ„ л¬ҙл°©
    */
    while (!q.empty()) {
        /*extract min*/
        int select_key = q.begin()->second;
        int min_of_key = q.begin()->first;
        q.erase(q.begin());

        if (selected[select_key]) {
            std::cout << " NOT MST" << std::endl;
            exit(1);
        }

        sum += min_of_key;
        selected[select_key] = true;
        std::cout << "dest : " << select_key << " (dis : " << vertex_key[select_key] << ")" << std::endl;

        /*decrease key*/
        for (auto e : adjList[select_key]) {
            if (!selected[e.second] && vertex_key[e.second] > e.first + vertex_key[select_key]) {
                q.erase({vertex_key[e.second], e.second});  //к°ҷмқҖ л…ёл“ңлЎң н–Ҙн•ҳлҠ” к°„м„ мӨ‘ weightк°Җ лҚ” мһ‘мқҖ к°„м„ мқҙ мһҲлӢӨл©ҙ к·ё м „ к°„м„ мқҖ мӮӯм ң
                q.insert({e.first, e.second});  //нҒҗм—җ мӮҪмһ…
                vertex_key[e.second] = e.first + vertex_key[select_key];
            }
        }
    }

    std::cout << std::endl;
    return sum;
}

void make_adj_list(std::vector<Graph> g, std::vector<std::vector<II>> &adj) {
    adj.resize(V);
    bool isEdge;
    for (int i = 0; i < g.size(); i++) {
        isEdge = false;
        int src = g[i].getSrc();
        int dest = g[i].getDest();
        int weight = g[i].getWeight();

        /*лҸҷмқј vertexлЎң н–Ҙн•ҳлҠ” к°„м„ мӨ‘ к°ҖмһҘ мһ‘мқҖ к°’л§Ңк°Җм§Җкі  мқём ‘ лҰ¬мҠӨнҠёлҘј л§Ңл“Өкё° мң„н•ң мҪ”л“ң*/
        if (adj[src].empty()) {
            adj[src].push_back({weight, dest});
        } else {
            for (int j = 0; j < adj[src].size(); j++) {
                if (adj[src][j].second == dest) {
                    isEdge = true;
                    if (adj[src][j].first > weight) {
                        adj[src][j].first = weight;
                    }
                }
            }
            if (!isEdge) adj[src].push_back({weight, dest});
        }

        isEdge = false;
        if (adj[dest].empty()) {
            adj[dest].push_back({weight, src});
        } else {
            for (int j = 0; j < adj[dest].size(); j++) {
                if (adj[dest][j].second == src) {
                    isEdge = true;
                    if (adj[dest][j].first > weight) {
                        adj[dest][j].first = weight;
                    }
                }
            }
            if (!isEdge) adj[dest].push_back({weight, src});
        }
    }
}

int prim_adjMatrix(std::vector<Graph> &g, std::vector<std::vector<int>> adjMatrix, int start) {
    int sum = 0;
    std::vector<int> vertex_key(V, INFINITY);     // vertexмқҳ мөңмҶҢ weightк°’ кі„мӮ°
    std::vector<bool> selected(g.size(), false);  //м„ нғқлҗң vertexмқёк°Җ

    vertex_key[start] = 0;  //мӢңмһ‘л…ёл“ң keyк°’ 0мңјлЎң мӢңмһ‘
    std::cout << "\nroute";
    /*vertex мҲҳл§ҢнҒј л°ҳліөн•ңлӢӨ*/
    for (int i = 0; i < V; i++) {
        int select_idx = -1;     //мқём ‘ vertexмӨ‘ к°ҖмһҘ мһ‘мқҖ к°ҖмӨ‘м№ҳлҘј к°–лҠ” vertex
        int min_key = INFINITY;  //мқём ‘ vertexмӨ‘ к°ҖмһҘ мһ‘мқҖ к°ҖмӨ‘м№ҳ

        /* мқём ‘ vertexмӨ‘ к°ҖмһҘ мһ‘мқҖ к°ҖмӨ‘м№ҳлҘј к°–лҠ” vertex м°ҫкё°*/
        for (int j = 0; j < V; j++) {
            if (!selected[j] && (min_key > vertex_key[j])) {
                select_idx = j;
                min_key = vertex_key[j];
            }
        }

        /*нҳ„мһ¬ мҪ”л“ңм—җм„ңлҠ” м—°кІ°м•Ҳлҗң к·ёлһҳн”„лҠ” мЈјм–ҙм§Җм§Җ м•Ҡкё° л•Ңл¬ём—җ
          м—Ҷм–ҙлҸ„ л¬ҙл°©н•ҳм§Җл§Ң л§Ңм•Ҫмқ„ мң„н•ң м—җлҹ¬мІҳлҰ¬*/
        if (select_idx == -1) {
            std::cout << "Not MST" << std::endl;
            exit(1);
        }

        selected[select_idx] = true;
        sum += min_key;
        std::cout << " -> " << select_idx << "(cost : " << min_key << ")";

        /*мқём ‘ vertexмқҳ weightк°Җ vertex_keyк°’ліҙлӢӨ мһ‘лӢӨл©ҙ keyк°’ к°ұмӢ  */
        for (int j = 0; j < V; j++) {
            if (!selected[j] && vertex_key[j] > adjMatrix[select_idx][j]) {
                vertex_key[j] = adjMatrix[select_idx][j];
            }
        }
    }
    std::cout << std::endl;
    return sum;
}

void make_adj_matrix(std::vector<Graph> g, std::vector<std::vector<int>> &adj) {
    adj.assign(V, std::vector<int>(V, INFINITY));
    for (int i = 0; i < g.size(); i++) {
        int src = g[i].getSrc();
        int dest = g[i].getDest();
        int weight = g[i].getWeight();

        if (adj[src][dest] > weight) {
            adj[src][dest] = weight;
        }
        if (adj[dest][src] > weight) {
            adj[dest][src] = weight;
        }
    }
}
